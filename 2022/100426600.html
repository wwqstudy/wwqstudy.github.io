mysql

命令行查看mysql账号密码

mysql -u root -p
show databases;
use 数据库名;
select * from users;sd
exit #退出数据库

mysql远程连接

mysql -u root -p(登录mysql)
use `mysql`;
grant all privileges on *.* to 'root'@'%' identified by 'root' WITH GRANT OPTION;
(
GRANT ALL PRIVILEGES ON *.* TO 'myuser'@'your ip' IDENTIFIED BY 'mypassword' WITH GRANT OPTION;
FLUSH PRIVILEGES;
)
flush privileges;

mysql建表查表操作

//查看当前数据库的所有表名
show tables;
show tables from 数据库名;
//在当前数据库中添加表
create table 表名(
//列名 字符格式 约束//注意末尾加逗号，最后一列末尾除外
uid int(32) primary,
uname varchar(32) ,
upassword varchar(32)
);
//查看表的结构
desc 表名;
//添加表列
//alter table 表名 add 字段 字符格式 约束
alter table user add unifo varchar(32) not null;
//修改表列
alter table 表名 modify 列名 字符格式 约束
//修改表名
alter table 表名  change 旧列名 新列名 新字符格式 新约束
//重命名
rename table 旧表名 to 新表名
//插入单行数据
insert into 表名 VALUES('1','java','34.4','7','张三');
//插入多行数据
insert into book(列名) VALUES('1','java','34.4','7','张三');

MobaXterm连接服务器：Network error:Connection refused

1.使用apt安装时报错：(可能在排查错误时出现)

E: Could not get lock /var/lib/dpkg/lock-frontend - open (11: Resource temporarily unavailable)
E: Unable to acquire the dpkg frontend lock (/var/lib/dpkg/lock-frontend), is another process using it?

解决方案

sudo killall apt apt-get

如果提示没有apt进程：

apt: no process found
apt-get: no process found

依次执行：

sudo rm /var/lib/apt/lists/lock
sudo rm /var/cache/apt/archives/lock
sudo rm /var/lib/dpkg/lock*
sudo dpkg --configure -a
sudo apt update

apt安装报错解决！

2.排查MobaXterm出错路径

1. 查看sshd相关服务是否存在
apt-get install openssh-server
systemctl restart ssh

2. 查看22端口是否存在
ss -ntl

3.将密码拒绝远程访问改一下配置
命令：gedit /etc/ssh/sshd_config
将#PermitRootLogin prohibit password
改为
PermitRootLogin yes
命令重启服务：systemctl restart ssh

MobaXterm连接服务器：Network error:Connection refused问题解决！

ubuntu开启FTP服务

sudo apt-get install vsftpd
sudo vi /etc/vsftpd.conf

local_enable=YES
write_enable=YES//修改这两行为上述所示

sudo /etc/init.d/vsftpd restart//重启FTP服务

//注：如果还无法连接,使用以下操作
//将pam_service_name=vsftpd 修改为：pam_service_name=ftp
//之后重启：sudo service vsftpd restart

ubuntu防火墙、端口

sudo ufw status//查看防火墙的状态
apt-get install ufw//安装防火墙
sudo ufw enable //开启防火墙
sudo ufw allow 端口号//开放某端口
sudo ufw reload //重启防火墙生效
sudo netstat -tunlp | grep 端口号//查看某端口监听状态
sudo ufw delete allow 端口号//防火墙关闭普通端口
sudo ufw allow 8001/tcp//防火墙开放规定协议的端口
sudo ufw delete allow 8001/tcp//防火墙关闭规定协议的端口
sudo ufw allow from ip//开放指定ip所有操作
sudo ufw delete allow from ip//关闭指定ip所有操作
sudo ufw allow from ip to any port 端口号//开放指定ip对应端口操作
sudo ufw delete allow from ip to any port 端口号//关闭指定ip对应端口操作
sudo ufw disable//关闭防火墙

ubuntu查看端口是否开放

netstat -aptn
iptables -nl//查看已开放端口
